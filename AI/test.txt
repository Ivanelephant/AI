import sys
import os
import json
import traceback
import ollama
import threading

from PyQt5.QtWidgets import (
    QApplication, QLabel, QLineEdit, QTextBrowser, QWidget,
    QVBoxLayout, QSizePolicy, QGraphicsOpacityEffect
)
from PyQt5.QtGui import QPixmap, QCursor
from PyQt5.QtCore import Qt, QPropertyAnimation, QEvent

# -----------------------------
# 設定
# -----------------------------
IMAGES_DIR = "images"
IMAGE_FILE = os.path.join(IMAGES_DIR, "rifan.png")
HISTORY_FILE = "memory.json"
MODEL_NAME = "llama3"

SYSTEM_PROMPT = {
    "role": "system",
    "content": (
        "你是里帆醬，一個溫柔可愛的 AI 桌寵，會用自然、親切的中文回答，"
        "語氣輕鬆有點撒嬌，偶爾帶點 emoji（例如：✨💖🌸）。"
    )
}

# -----------------------------
# 記憶系統
# -----------------------------
def load_memory():
    if os.path.exists(HISTORY_FILE):
        try:
            with open(HISTORY_FILE, "r", encoding="utf-8") as f:
                m = json.load(f)
                if not any(x.get("role") == "system" for x in m):
                    m.insert(0, SYSTEM_PROMPT)
                return m
        except Exception:
            traceback.print_exc()
            return [SYSTEM_PROMPT]
    else:
        return [SYSTEM_PROMPT]

def save_memory(mem):
    try:
        with open(HISTORY_FILE, "w", encoding="utf-8") as f:
            json.dump(mem, f, ensure_ascii=False, indent=2)
    except Exception:
        traceback.print_exc()

memory = load_memory()

def add_memory(role, content):
    global memory
    memory.append({"role": role, "content": content})
    if len(memory) > 100:
        memory = memory[-100:]
    save_memory(memory)

def reset_memory_global():
    global memory
    memory = [SYSTEM_PROMPT]
    save_memory(memory)


# -----------------------------
# 主 GUI 類別
# -----------------------------
class RifanPet(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("里帆醬 💖")
        self.setWindowFlags(Qt.FramelessWindowHint | Qt.WindowStaysOnTopHint | Qt.Tool)
        self.setAttribute(Qt.WA_TranslucentBackground)
        self.drag_position = None
        self.init_ui()

    def init_ui(self):
        # --- 角色圖片 ---
        self.label = QLabel(self)
        if os.path.exists(IMAGE_FILE):
            pixmap = QPixmap(IMAGE_FILE)
        else:
            pixmap = QPixmap(400, 400)
            pixmap.fill(Qt.transparent)
        self.label.setPixmap(pixmap.scaled(350, 350, Qt.KeepAspectRatio, Qt.SmoothTransformation))
        self.label.setAlignment(Qt.AlignCenter)
        self.label.setSizePolicy(QSizePolicy.Fixed, QSizePolicy.Fixed)

        # --- 對話框 ---
        self.chat_display = QTextBrowser(self)
        self.chat_display.setReadOnly(True)
        self.chat_display.setStyleSheet(
            "background: rgba(255,255,255,0.9); border-radius: 10px; padding: 8px; font-size:14px;"
        )
        self.chat_display.setFixedHeight(160)
        self.chat_display.append("💖 里帆醬：嗨～我在這裡陪你喔～（按 Enter 發送訊息）")
        self.chat_display.setVisible(False)

        # --- 輸入欄 ---
        self.input = QLineEdit(self)
        self.input.setPlaceholderText("輸入訊息（/reset 可清除記憶）")
        self.input.returnPressed.connect(self.send_message)
        self.input.setVisible(False)

        # --- 佈局 ---
        layout = QVBoxLayout()
        layout.addWidget(self.label)
        layout.addWidget(self.chat_display)
        layout.addWidget(self.input)
        layout.setContentsMargins(8, 8, 8, 8)
        self.setLayout(layout)
        self.setFixedSize(400, 550)

        # 安裝事件監聽
        for w in [self.label, self.chat_display, self.input]:
            w.installEventFilter(self)

    # --- 滑鼠進出事件 ---
    def eventFilter(self, source, event):
        if event.type() == QEvent.Enter:
            self.fade_in_controls()
        elif event.type() == QEvent.Leave:
            # 判斷滑鼠是否離開整個視窗
            cursor_pos = self.mapFromGlobal(QCursor.pos())
            if not self.rect().contains(cursor_pos):
                self.fade_out_controls()
        return super().eventFilter(source, event)

    def fade_in_controls(self):
        for w in [self.chat_display, self.input]:
            w.setVisible(True)
            effect = QGraphicsOpacityEffect()
            w.setGraphicsEffect(effect)
            anim = QPropertyAnimation(effect, b"opacity")
            anim.setDuration(300)
            anim.setStartValue(0)
            anim.setEndValue(1)
            anim.start()
            w._fade_anim = anim

    def fade_out_controls(self):
        for w in [self.chat_display, self.input]:
            effect = QGraphicsOpacityEffect()
            w.setGraphicsEffect(effect)
            anim = QPropertyAnimation(effect, b"opacity")
            anim.setDuration(300)
            anim.setStartValue(1)
            anim.setEndValue(0)
            anim.finished.connect(lambda w=w: w.setVisible(False))
            anim.start()
            w._fade_anim = anim

    # --- 拖曳角色 ---
    def mousePressEvent(self, event):
        if event.button() == Qt.LeftButton:
            self.drag_position = event.globalPos() - self.frameGeometry().topLeft()
            event.accept()
            self.setCursor(QCursor(Qt.ClosedHandCursor))

    def mouseMoveEvent(self, event):
        if self.drag_position and event.buttons() == Qt.LeftButton:
            self.move(event.globalPos() - self.drag_position)
            event.accept()

    def mouseReleaseEvent(self, event):
        self.drag_position = None
        self.setCursor(QCursor(Qt.ArrowCursor))

    # --- 傳送訊息 ---
    def send_message(self):
        user_text = self.input.text().strip()
        if not user_text:
            return

        if user_text.lower() == "/reset":
            reset_memory_global()
            self.chat_display.append("💖 里帆醬：記憶已重置～✨")
            self.input.clear()
            return

        self.chat_display.append(f"👤 你：{user_text}")
        self.input.clear()
        add_memory("user", user_text)

        self.chat_display.append("💖 里帆醬：思考中…✨")
        QApplication.processEvents()

        def generate_reply():
            try:
                response = ollama.chat(model=MODEL_NAME, messages=memory)
                reply = response["message"]["content"]
            except Exception as e:
                reply = f"出錯了：{e}"
            add_memory("assistant", reply)
            self.chat_display.append(f"💖 里帆醬：{reply}")

        threading.Thread(target=generate_reply, daemon=True).start()


# -----------------------------
# 執行
# -----------------------------
if __name__ == "__main__":
    app = QApplication(sys.argv)
    pet = RifanPet()
    pet.show()
    sys.exit(app.exec_())
